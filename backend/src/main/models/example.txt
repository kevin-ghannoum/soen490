'use strict';
import { Table, Column, Model, HasMany, PrimaryKey, DataType } from 'sequelize-typescript'

@Table
class Example extends Model {   
    
    @PrimaryKey --> If primary key is not marked, then it will created an integer and increment it ("id")
    @Column
    id!: number

    @Column
    name!: string

    //If type should not or cannot be inferred
    @Column(DataType.TEXT)
    randomText!: string

};
